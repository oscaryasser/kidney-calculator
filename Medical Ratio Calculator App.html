            const [urineCr, setUrineCr] = useState('');

            const [krRatio, setKrRatio] = useState(null);

            const [krInterpretation, setKrInterpretation] = useState('');


            // Fractional Excretion K State

            const [fekUrineK, setFekUrineK] = useState('');

            const [fekUrineCr, setFekUrineCr] = useState('');

            const [fekSerumK, setFekSerumK] = useState('');

            const [fekSerumCr, setFekSerumCr] = useState('');

            const [fekInterpretation, setFekInterpretation] = useState('');


            // Generic Fractional Excretion State

            const [substance, setSubstance] = useState('Sodium');

            const [urineX, setUrineX] = useState('');

            const [genericUrineCr, setGenericUrineCr] = useState('');

            const [serumX, setSerumX] = useState('');

            const [genericSerumCr, setGenericSerumCr] = useState('');

            const [genericResult, setGenericResult] = useState(null);


            // Calculate K/Cr Ratio

            const calculateKCrRatio = () => {

                if (!urineK || !urineCr) return;

                

                const k = parseFloat(urineK);

                const cr = parseFloat(urineCr);

                

                // Convert urine creatinine from mg/dL to mmol/L

                // Multiply by 88 to get μmol/L, then divide by 1000 to get mmol/L

                const crConverted = (cr * 88) / 1000;

                

                const ratio = k / crConverted;

                setKrRatio(ratio.toFixed(2));

                

                // Interpretation based on the reference values

                if (ratio > 2.5) {

                    setKrInterpretation('High (>2.5) - Renal potassium wasting (diuretics, hyperaldosteronism, vomiting, metabolic alkalosis)');

                } else if (ratio < 2.5) {

                    setKrInterpretation('Low (<2.5) - Potassium conservation (extra-renal losses, decreased intake, cellular shift)');

                }

            };


            // Calculate Fractional Excretion of K

            const calculateFEK = () => {

                if (!fekUrineK || !fekUrineCr || !fekSerumK || !fekSerumCr) return;

                

                const uK = parseFloat(fekUrineK);

                const uCr = parseFloat(fekUrineCr);

                const sK = parseFloat(fekSerumK);

                const sCr = parseFloat(fekSerumCr);

                

                const fe = ((uK * sCr) / (sK * uCr)) * 100;

                setFekResult(fe.toFixed(1));

                

                // Interpretation based on reference values

                if (fe > 9.5) {

                    setFekInterpretation('High (>9.5%, mean 15%) - Renal potassium wasting (diuretics, Bartter/Gitelman, vomiting, metabolic alkalosis)');

                } else if (fe >= 4 && fe <= 16) {

                    setFekInterpretation('Normal (4-16%, mean 8%) - Normal potassium handling');

                } else if (fe < 6.4) {

                    setFekInterpretation('Low (<6.4%, mean 3%) - Renal potassium conservation (extra-renal losses, decreased intake, cellular shift)');

                }

            };


            // Calculate Generic Fractional Excretion

            const calculateGenericFE = () => {

                if (!urineX || !genericUrineCr || !serumX || !genericSerumCr) return;

                

                const uX = parseFloat(urineX);

                const uCr = parseFloat(genericUrineCr);

                const sX = parseFloat(serumX);

                const sCr = parseFloat(genericSerumCr);

                

                const fe = ((uX * sCr) / (sX * uCr)) * 100;

                setGenericResult(fe.toFixed(2));

            };

                        setFekSerumCr('');

                        setFekInterpretation('');

                        setSerumX('');

                        break;

                }
